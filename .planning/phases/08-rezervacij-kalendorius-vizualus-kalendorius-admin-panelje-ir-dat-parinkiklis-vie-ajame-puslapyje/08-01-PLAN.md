**4. Add `renderCalendar()` function** — This is the core rendering logic:

```javascript
const LT_MONTHS = ['Sausis','Vasaris','Kovas','Balandis','Geguzė','Birželis','Liepa','Rugpjūtis','Rugsėjis','Spalis','Lapkritis','Gruodis'];

function renderCalendar() {
  const grid = document.getElementById('calendar-grid');
  const label = document.getElementById('cal-month-label');
  if (!grid || !label) return;

  label.textContent = `${LT_MONTHS[calendarMonth]} ${calendarYear}`;

  // Remove old day cells (keep 7 header divs)
  const headers = grid.querySelectorAll('.cal-day-header');
  grid.querySelectorAll('.cal-day').forEach(el => el.remove());

  // Build day grid
  const firstDay = new Date(calendarYear, calendarMonth, 1);
  const lastDay = new Date(calendarYear, calendarMonth + 1, 0);
  const startDow = (firstDay.getDay() + 6) % 7; // Monday=0
  const daysInMonth = lastDay.getDate();

  const today = new Date();
  const todayStr = today.toISOString().split('T')[0];

  // Build reservation lookup: dateStr -> [reservation, ...]
  const dayMap = {};
  reservations.forEach(r => {
    if (!r.arrival_date || !r.departure_date) return;
    const start = new Date(r.arrival_date);
    const end = new Date(r.departure_date);
    for (let d = new Date(start); d < end; d.setDate(d.getDate() + 1)) {
      const key = d.toISOString().split('T')[0];
      if (!dayMap[key]) dayMap[key] = [];
      dayMap[key].push(r);
    }
  });

  // Empty cells before month starts
  for (let i = 0; i < startDow; i++) {
    const cell = document.createElement('div');
    cell.className = 'cal-day other-month';
    grid.appendChild(cell);
  }

  // Day cells
  for (let day = 1; day <= daysInMonth; day++) {
    const dateStr = `${calendarYear}-${String(calendarMonth + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const cell = document.createElement('div');
    cell.className = 'cal-day' + (dateStr === todayStr ? ' today' : '');

    const numEl = document.createElement('div');
    numEl.className = 'cal-day-number';
    numEl.textContent = day;
    cell.appendChild(numEl);

    // Reservation chips for this day
    const dayReservations = dayMap[dateStr] || [];
    dayReservations.forEach(r => {
      const chip = document.createElement('div');
      chip.className = `calendar-chip status-${r.status}`;
      const roomLabel = r.hotel_rooms?.name || ('K' + (r.hotel_rooms?.number || '?'));
      chip.textContent = `${r.guest_name} · ${roomLabel}`;
      chip.title = `${r.guest_name} — ${r.arrival_date} → ${r.departure_date} (${r.status})`;
      chip.addEventListener('click', () => showReservationFromCalendar(r.id));
      cell.appendChild(chip);
    });

    grid.appendChild(cell);
  }
}
```

```javascript
function showReservationFromCalendar(reservationId) {
  // Switch to list view
  document.querySelectorAll('.res-view-tab').forEach(t => t.classList.remove('active'));
  const listTab = document.querySelector('.res-view-tab[data-view="list"]');
  if (listTab) listTab.classList.add('active');
  document.getElementById('res-view-list').style.display = '';
  document.getElementById('res-view-calendar').style.display = 'none';

  // Set filter to "all" so the reservation is visible
  activeFilter = 'all';
  renderReservationsSection();

  // Scroll to and highlight the reservation card
  setTimeout(() => {
    const card = document.querySelector(`.reservation-card [data-id="${reservationId}"]`);
    if (card) {
      const cardEl = card.closest('.reservation-card');
      if (cardEl) {
        cardEl.scrollIntoView({ behavior: 'smooth', block: 'center' });
        cardEl.style.outline = '2px solid #FFC700';
        setTimeout(() => { cardEl.style.outline = ''; }, 2000);
      }
    }
  }, 100);
}
```
