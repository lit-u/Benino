This file is the dispatch router — it maps each `source.type` value from config.json to its fetcher function. There are 7 type values: rss, hf-papers-api, hf-models-api, hn-algolia, github-trending, github-search, taaft-scrape.

```javascript
import { fetchRss } from './rss-fetcher.js';
import { fetchHackerNews } from './hn-fetcher.js';
import { fetchGithubTrending } from './github-trending-fetcher.js';
import { fetchGithubSearch } from './github-search-fetcher.js';
import { fetchHuggingFacePapers, fetchHuggingFaceModels } from './huggingface-fetcher.js';
import { fetchTaaft } from './taaft-fetcher.js';

const FETCHERS = {
  'rss':             fetchRss,
  'hf-papers-api':   fetchHuggingFacePapers,
  'hf-models-api':   fetchHuggingFaceModels,
  'hn-algolia':      fetchHackerNews,
  'github-trending': fetchGithubTrending,
  'github-search':   fetchGithubSearch,
  'taaft-scrape':    fetchTaaft
};

/**
 * Fetch items from a single source config entry.
 * Throws if no fetcher is registered for source.type.
 */
export async function fetchSource(source) {
  const fetcher = FETCHERS[source.type];
  if (!fetcher) {
    throw new Error(`No fetcher registered for source type: "${source.type}" (source id: ${source.id})`);
  }
  return fetcher(source);
}
```

```javascript
import { createRequire } from 'module';
import { fileURLToPath } from 'url';
import path from 'path';
import cron from 'node-cron';
import { fetchSource } from './sources/index.js';
import { initDb, isSeen, markSeen } from './storage/db.js';

// JSON import with createRequire (works reliably in all Node.js ESM versions)
const require = createRequire(import.meta.url);
const config = require('./config.json');

export function registerCron() {
  const expr = config.pollIntervalCron;
  cron.schedule(expr, runCollector);
  console.log(`[news-collector] Cron scheduled: ${expr}`);
}

export async function runCollector() {
  const db = initDb();
  const startedAt = Date.now();
  console.log(`[news-collector] Run started ${new Date().toISOString()}`);

  const results = { fetched: 0, new: 0, skipped: 0, errors: [] };

  const enabledSources = config.sources.filter(s => s.enabled);

  for (const source of enabledSources) {
    try {
      const items = await fetchSource(source);
      for (const item of items) {
        results.fetched++;
        if (isSeen(db, item.url)) {
          results.skipped++;
        } else {
          markSeen(db, item.url, item.title, source.id);
          results.new++;
        }
      }
    } catch (err) {
      results.errors.push({ source: source.id, error: err.message });
      console.error(`[news-collector] Error fetching ${source.id}: ${err.message}`);
    }
  }

  const duration = Date.now() - startedAt;
  console.log(
    `[news-collector] Done in ${duration}ms — ` +
    `fetched:${results.fetched} new:${results.new} skipped:${results.skipped} errors:${results.errors.length}`
  );

  db.close();
  return results;
}
```
