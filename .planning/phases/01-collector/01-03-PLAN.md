```javascript
import fetch from 'node-fetch';

export async function fetchGithubSearch(source) {
  const since = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString().split('T')[0];
  const query = source.query.replace('DATE', since);
  const url = `https://api.github.com/search/repositories?q=${encodeURIComponent(query)}&sort=${source.sort}&per_page=${source.perPage}&order=desc`;

  const headers = {
    'Accept': 'application/vnd.github+json',
    'X-GitHub-Api-Version': '2022-11-28'
  };

  if (process.env.GITHUB_TOKEN) {
    headers['Authorization'] = `Bearer ${process.env.GITHUB_TOKEN}`;
  }

  const res = await fetch(url, { headers });

  const remaining = res.headers.get('X-RateLimit-Remaining');
  if (remaining !== null && parseInt(remaining) < 2) {
    console.warn('[github-search] Rate limit nearly exhausted, remaining:', remaining);
  }

  if (res.status === 403 || res.status === 429) {
    throw new Error(`GitHub Search rate limited: ${res.status}`);
  }

  if (!res.ok) {
    throw new Error(`GitHub Search returned ${res.status}`);
  }

  const data = await res.json();

  return (data.items || []).map(repo => ({
    url: repo.html_url,
    title: repo.full_name,
    publishedAt: repo.updated_at || '',
    source: source.id,
    summary: (repo.description || '').substring(0, 500),
    raw: repo
  }));
}
```

```javascript
import fetch from 'node-fetch';

export async function fetchHuggingFacePapers(source) {
  const res = await fetch(
    `https://huggingface.co/api/daily_papers?limit=${source.limit || 20}`,
    { headers: { 'User-Agent': 'NewsBot/1.0' } }
  );

  if (!res.ok) {
    throw new Error(`HuggingFace Papers API returned ${res.status}`);
  }

  const papers = await res.json();

  return (Array.isArray(papers) ? papers : []).map(p => ({
    url: `https://huggingface.co/papers/${p.paper?.id || p.id}`,
    title: p.paper?.title || p.title || '',
    publishedAt: p.publishedAt || '',
    source: source.id,
    summary: (p.summary || p.paper?.summary || '').substring(0, 500),
    raw: p
  })).filter(i => i.url && !i.url.endsWith('/papers/'));
}

export async function fetchHuggingFaceModels(source) {
  const params = new URLSearchParams({
    sort: 'createdAt',      // CRITICAL: camelCase — 'created_at' returns HTTP 400
    direction: '-1',
    limit: String(source.limit || 20)
  });
  if (source.filter) params.set('filter', source.filter);

  const res = await fetch(
    `https://huggingface.co/api/models?${params}`,
    { headers: { 'User-Agent': 'NewsBot/1.0' } }
  );

  if (!res.ok) {
    throw new Error(`HuggingFace Models API returned ${res.status}`);
  }

  const models = await res.json();

  return (Array.isArray(models) ? models : []).map(m => ({
    url: `https://huggingface.co/${m.id}`,
    title: m.id || '',
    publishedAt: m.createdAt || '',
    source: source.id,
    summary: (m.description || '').substring(0, 500),
    raw: m
  })).filter(i => i.url && i.title);
}
```

```javascript
import fetch from 'node-fetch';
import * as cheerio from 'cheerio';

export async function fetchTaaft(source) {
  const res = await fetch(source.url, {
    headers: {
      'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
    }
  });

  if (!res.ok) {
    throw new Error(`TAAFT returned ${res.status}`);
  }

  const html = await res.text();
  const $ = cheerio.load(html);

  const items = [];
  // REPLACE CORRECT_SELECTOR with the selector identified in Step 1
  $('CORRECT_SELECTOR').each((_, el) => {
    // Extract the tool link — find the <a> with href pointing to a tool
    const linkEl = $(el).is('a') ? $(el) : $(el).find('a[href]').first();
    const href = linkEl.attr('href');
    if (!href) return;

    const url = href.startsWith('http') ? href : `https://theresanaiforthat.com${href}`;
    const title = linkEl.text().trim() || $(el).find('h2,h3,h4,[class*=title],[class*=name]').first().text().trim() || '';
    if (!title || !url) return;

    const summary = $(el).find('p,[class*=description],[class*=summary]').first().text().trim().substring(0, 500);

    items.push({
      url,
      title,
      publishedAt: new Date().toISOString(),
      source: source.id,
      summary,
      raw: { html: $(el).html()?.substring(0, 200) }
    });
  });

  // Deduplicate by URL within this batch
  const seen = new Set();
  return items.filter(i => {
    if (seen.has(i.url)) return false;
    seen.add(i.url);
    return true;
  });
}
```
