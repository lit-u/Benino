<!-- API contract from Plan 01 -->

POST /api/hotels/:slug/rooms — Create room
  Body: { number, name, price_per_night, area_m2, sea_distance_m, orientation, orientation_note, noise_level, availability_status, available_from, description }
  Response: { success: true, room: {...} }

PUT /api/hotels/:slug/rooms/:roomId — Update room
  Body: same fields as POST
  Response: { success: true, room: {...} }

DELETE /api/hotels/:slug/rooms/:roomId
  Response: { success: true }

POST /api/hotels/:slug/rooms/:roomId/photos — Upload photos
  FormData: photos (files), type ('window_view' | 'interior')
  Response: { success: true, photos: [...] }

PUT /api/hotels/:slug/rooms/:roomId/photos/order — Reorder photos
  Body: { order: [{ id, sort_order }, ...] }
  Response: { success: true }

DELETE /api/hotels/:slug/rooms/:roomId/photos/:photoId
  Response: { success: true }

GET /api/hotels/:slug/rooms/:num/qr — QR code PNG
  Response: image/png binary

GET /api/hotels/:slug — Full hotel with rooms and photos
  Response: { success: true, hotel: { ...hotel, hotel_rooms: [{ ...room, room_photos: [...] }] } }

Auth: All mutation endpoints require X-Session-ID header. Server checks owner_id match.
Admin route: GET /hotel/:slug/admin uses requireUser middleware (returns 401 without auth).

**Drag-drop implementation (from RESEARCH.md):**
```javascript
function initDragDropReorder(container) {
  let dragSrcEl = null;
  container.querySelectorAll('.photo-item').forEach(item => {
    item.draggable = true;
    item.addEventListener('dragstart', (e) => {
      dragSrcEl = item;
      item.classList.add('dragging');
      e.dataTransfer.effectAllowed = 'move';
    });
    item.addEventListener('dragover', (e) => {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
      const target = e.currentTarget;
      if (target !== dragSrcEl) {
        const rect = target.getBoundingClientRect();
        const after = e.clientX > rect.left + rect.width / 2;
        container.insertBefore(dragSrcEl, after ? target.nextSibling : target);
      }
    });
    item.addEventListener('dragend', () => {
      item.classList.remove('dragging');
      clearTimeout(window._reorderTimer);
      window._reorderTimer = setTimeout(() => savePhotoOrder(container), 300);
    });
  });
}
```
