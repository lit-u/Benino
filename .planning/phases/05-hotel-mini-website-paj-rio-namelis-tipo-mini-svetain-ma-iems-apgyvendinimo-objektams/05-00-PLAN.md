```javascript
import { test, expect } from '@playwright/test';

test.describe('Hotel Mini-Website', () => {

  // HOTEL-01: SSR title
  test.skip('ssr-title: /hotel/:slug returns HTML with hotel name in title', async ({ page }) => {
    // Requires: Plan 01 (API) + Plan 02 (SSR route) + test hotel in DB
    await page.goto('/hotel/test-hotel');
    await expect(page).toHaveTitle(/Test Hotel.*Pajūrio Portalas/);
  });

  // HOTEL-02: JSON-LD
  test.skip('json-ld: page source contains LodgingBusiness schema', async ({ page }) => {
    // Requires: Plan 02 (SSR route with buildLodgingSchema)
    await page.goto('/hotel/test-hotel');
    const jsonLd = await page.locator('script[type="application/ld+json"]').textContent();
    expect(jsonLd).toContain('LodgingBusiness');
  });

  // HOTEL-03: CRUD API
  test.skip('create-hotel: POST /api/hotels creates hotel and returns slug', async ({ request }) => {
    // Requires: Plan 01 (hotels.js route + server wiring)
    const response = await request.post('/api/hotels', {
      data: { name: 'Test Hotel', city: 'Palanga', address: 'Jūros g. 1' },
      headers: { 'X-Session-ID': 'test-session' }
    });
    expect(response.status()).toBe(201);
    const body = await response.json();
    expect(body.success).toBe(true);
    expect(body.slug).toBeTruthy();
  });

  // HOTEL-04: Admin auth gate
  test.skip('admin-auth: GET /hotel/:slug/admin returns 401 without session', async ({ request }) => {
    // Requires: Plan 02 (admin route with requireUser middleware)
    const response = await request.get('/hotel/test-hotel/admin');
    expect(response.status()).toBe(401);
  });

  // HOTEL-05: Photo upload
  test.skip('photo-upload: POST photos returns success with URL', async ({ request }) => {
    // Requires: Plan 01 (API + image service) + test hotel + test room in DB
    // This test needs a real hotel/room ID — will be parameterized when backend exists
    const response = await request.post('/api/hotels/test-hotel/rooms/test-room-id/photos', {
      headers: { 'X-Session-ID': 'test-session' },
      multipart: {
        type: 'interior',
        photos: { name: 'test.jpg', mimeType: 'image/jpeg', buffer: Buffer.from('fake-image') }
      }
    });
    // Expect either 201 (success) or appropriate error for fake image
    expect([201, 400]).toContain(response.status());
  });

  // HOTEL-06: QR PNG
  test.skip('qr-png: GET /api/hotels/:slug/rooms/:num/qr returns PNG', async ({ request }) => {
    // Requires: Plan 01 (QR endpoint)
    const response = await request.get('/api/hotels/test-hotel/rooms/1/qr');
    expect(response.status()).toBe(200);
    expect(response.headers()['content-type']).toContain('image/png');
  });

});
```

All tests use `test.skip()` so the file compiles and runs (showing skipped) without requiring the backend to exist. As each plan is completed, the corresponding `test.skip()` should be changed to `test()` by the executor.
