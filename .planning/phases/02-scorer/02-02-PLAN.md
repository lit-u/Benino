// scorer/heuristics.js
// Fast heuristic scoring — no I/O, no dependencies, pure function.
// Returns: { score: number, type: 'breakthrough'|'release'|'update'|'research' }

const SOURCE_BASELINES = {
  'anthropic':          { base: 60, typeHint: 'release' },
  'openai':             { base: 60, typeHint: 'release' },
  'google-ai':          { base: 60, typeHint: 'release' },
  'huggingface-papers': { base: 50, typeHint: 'research' },
  'huggingface-models': { base: 40, typeHint: 'update' },
  'hackernews':         { base: 40, typeHint: 'update' },
  'github-trending':    { base: 35, typeHint: 'release' },
  'github-search':      { base: 35, typeHint: 'update' },
  'github-changelog':   { base: 30, typeHint: 'update' },
  'taaft':              { base: 25, typeHint: 'update' },
};

// Keyword → { scoreBoost, type } — checked against lowercased title
const KEYWORD_RULES = [
  // Breakthrough signals — high boost, override type
  { words: ['gpt-5','gpt-6','claude 4','claude 3.7','gemini 2.5','llama 4','grok 3'], boost: 35, type: 'breakthrough' },
  { words: ['agi','superhuman','surpasses human','artificial general intelligence'], boost: 30, type: 'breakthrough' },
  { words: ['acqui','acquired','billion dollar','raises $','series a','series b','series c'], boost: 25, type: 'breakthrough' },
  { words: ['launches','introduces','announces','new model','release of','releasing'], boost: 20, type: 'breakthrough' },
  { words: ['state-of-the-art','sota','beats','outperforms','world record','surpasses'], boost: 20, type: 'breakthrough' },
  // Release signals
  { words: ['released','now available','open source','open-source','v2.','v3.','version 2','version 3'], boost: 10, type: 'release' },
  { words: ['available','launched','published','shipped','deployed'], boost: 5, type: 'release' },
  // Research signals
  { words: ['arxiv','we propose','we present','novel approach','benchmark','evaluation','study shows'], boost: 8, type: 'research' },
  { words: ['paper','research','academic','analysis','survey'], boost: 3, type: 'research' },
  // Negative signals — reduce score
  { words: ['bug fix','patch notes','minor update','deprecat','hotfix'], boost: -15, type: 'update' },
  { words: ['changelog','v1.0.1','v0.','typo','docs update'], boost: -10, type: 'update' },
];

export function scoreByHeuristics(item) {
  const { title = '', source_id = '' } = item;
  const t = title.toLowerCase();

  const baseline = SOURCE_BASELINES[source_id] ?? { base: 30, typeHint: 'update' };
  let score = baseline.base;
  let type = baseline.typeHint;

  // Apply keyword rules — highest-priority match wins the type; all boosts accumulate
  let highestBoost = 0;
  for (const rule of KEYWORD_RULES) {
    const matched = rule.words.some(w => t.includes(w));
    if (matched) {
      score += rule.boost;
      if (Math.abs(rule.boost) > Math.abs(highestBoost)) {
        highestBoost = rule.boost;
        if (rule.type) type = rule.type;
      }
    }
  }

  // Clamp to [0, 100]
  score = Math.max(0, Math.min(100, score));

  return { score, type };
}

// scorer/multi-source.js
// Detect same story across multiple sources within a batch.
// Applies score boost: +5 per extra source, cap +15.
// Mutates items array in place.
// No external dependencies — pure JS.

function normTitle(t) {
  return (t || '').toLowerCase().replace(/[^a-z0-9 ]/g, '').replace(/\s+/g, ' ').trim();
}

function jaccardSimilarity(a, b) {
  const sa = new Set(normTitle(a).split(' ').filter(Boolean));
  const sb = new Set(normTitle(b).split(' ').filter(Boolean));
  if (sa.size === 0 && sb.size === 0) return 1;
  if (sa.size === 0 || sb.size === 0) return 0;
  const intersection = [...sa].filter(w => sb.has(w)).length;
  const union = new Set([...sa, ...sb]).size;
  return intersection / union;
}

export function applyMultiSourceBoost(items) {
  // Build groups: items[i] and items[j] with Jaccard >= 0.5 are the same story
  const grouped = new Array(items.length).fill(-1); // group index per item
  let groupCount = 0;

  for (let i = 0; i < items.length; i++) {
    if (grouped[i] !== -1) continue; // already assigned
    grouped[i] = groupCount;
    for (let j = i + 1; j < items.length; j++) {
      if (grouped[j] !== -1) continue;
      const sim = jaccardSimilarity(items[i].title, items[j].title);
      if (sim >= 0.5) {
        grouped[j] = groupCount;
      }
    }
    groupCount++;
  }

  // Count group sizes
  const groupSizes = new Array(groupCount).fill(0);
  for (const g of grouped) groupSizes[g]++;

  // Apply boost: (groupSize - 1) * 5, capped at 15
  for (let i = 0; i < items.length; i++) {
    const g = grouped[i];
    const extraSources = groupSizes[g] - 1;
    if (extraSources > 0) {
      const boost = Math.min(extraSources * 5, 15);
      items[i].score = Math.min(100, items[i].score + boost);
    }
  }
}
