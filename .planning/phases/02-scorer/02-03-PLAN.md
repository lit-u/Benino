// scorer/llm-scorer.js
// LLM refinement for ambiguous items (heuristic score 30-70).
// Uses openai SDK with OpenRouter baseURL — no new dependencies.
// Graceful fallback: if OPENROUTER_API_KEY absent or API fails, returns empty array
// (caller keeps heuristic scores as-is).

import OpenAI from 'openai';

const SYSTEM_PROMPT = `You are a tech news importance scorer.
For each news item, return JSON with: score (0-100), type (breakthrough|release|update|research).

Scoring guide:
- 70-100: Major model launches, AGI announcements, billion-dollar acquisitions, landmark research
- 40-69: New tool releases, significant paper, notable open-source project, company news
- 10-39: Minor updates, changelogs, tool listings, routine research
- 0-9: Spam, duplicates, irrelevant

Types:
- breakthrough: New models (GPT-5, Claude 4, etc.), capability leaps, major research findings
- release: Software release, open-source launch, product launch
- update: Changelog, minor version, bug fix, feature addition
- research: ArXiv paper, academic study, technical report

Return ONLY valid JSON array: [{"url_hash":"...","score":85,"type":"breakthrough"}]`;

/**
 * Score a batch of ambiguous items via LLM.
 * @param {Array<{url_hash, title, url, source_id}>} items
 * @param {object} scoringCfg — from config.json scoring block
 * @returns {Promise<Array<{url_hash: string, score: number, type: string}>>}
 */
export async function scoreBatchWithLLM(items, scoringCfg) {
  const apiKey = process.env.OPENROUTER_API_KEY;
  if (!apiKey) {
    console.warn('[scorer/llm] OPENROUTER_API_KEY not set — skipping LLM scoring (heuristics-only mode)');
    return [];
  }

  const client = new OpenAI({
    baseURL: 'https://openrouter.ai/api/v1',
    apiKey,
  });

  const results = [];
  const batchSize = scoringCfg.llmBatchSize ?? 20;

  // Process in batches of batchSize
  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize);
    const itemsPayload = batch.map(item => ({
      url_hash: item.url_hash,
      title: item.title || '(no title)',
      source: item.source_id,
    }));

    const userMessage = `Score these ${batch.length} items:\n${JSON.stringify(itemsPayload, null, 2)}`;

    try {
      const response = await client.chat.completions.create({
        model: scoringCfg.llmModel ?? 'google/gemini-2.0-flash-001',
        messages: [
          { role: 'system', content: SYSTEM_PROMPT },
          { role: 'user', content: userMessage },
        ],
        temperature: 0.1,
        max_tokens: 600,
      });

      const raw = response.choices[0]?.message?.content ?? '';
      // Strip markdown fences if LLM wraps in ```json
      const cleaned = raw.replace(/```json\n?|\n?```/g, '').trim();

      let parsed;
      try {
        parsed = JSON.parse(cleaned);
      } catch (parseErr) {
        console.warn(`[scorer/llm] JSON parse failed for batch ${i}–${i + batchSize}: ${parseErr.message} — using heuristic scores`);
        continue; // skip this batch, heuristic scores remain
      }

      if (!Array.isArray(parsed)) {
        console.warn('[scorer/llm] LLM response is not an array — skipping batch');
        continue;
      }

      for (const entry of parsed) {
        if (entry.url_hash && typeof entry.score === 'number') {
          results.push({
            url_hash: entry.url_hash,
            score: Math.max(0, Math.min(100, entry.score)),
            type: ['breakthrough','release','update','research'].includes(entry.type)
              ? entry.type : 'update',
          });
        }
      }

      console.log(`[scorer/llm] Batch ${Math.floor(i/batchSize)+1}: scored ${parsed.length} items`);

    } catch (apiErr) {
      console.warn(`[scorer/llm] API error for batch ${i}–${i + batchSize}: ${apiErr.message} — using heuristic scores`);
      // Continue to next batch — graceful degradation
    }
  }

  return results;
}

```javascript
// scorer/index.js
// runScorer(db) — main scoring pipeline entry point.
// Called after each runCollector() pass.
// Scores all items where score IS NULL.

import { createRequire } from 'module';
import { scoreByHeuristics } from './heuristics.js';
import { scoreBatchWithLLM } from './llm-scorer.js';
import { applyMultiSourceBoost } from './multi-source.js';

const require = createRequire(import.meta.url);
const config = require('../config.json');

/**
 * Score all unscored items in the database.
 * @param {import('better-sqlite3').Database} db — must be returned by initDb() (columns already exist)
 * @returns {Promise<{scored: number}>}
 */
export async function runScorer(db) {
  const scoringCfg = config.scoring;
  if (!scoringCfg?.enabled) {
    console.log('[scorer] Scoring disabled in config — skipping');
    return { scored: 0 };
  }

  // Stage 0: Query unscored items
  const unscored = db.prepare(
    'SELECT url_hash, url, title, source_id FROM seen_urls WHERE score IS NULL'
  ).all();

  if (unscored.length === 0) {
    console.log('[scorer] No unscored items — nothing to do');
    return { scored: 0 };
  }

  console.log(`[scorer] Scoring ${unscored.length} items...`);

  // Stage 1: Heuristics on all items
  const withScores = unscored.map(item => ({
    ...item,
    ...scoreByHeuristics(item),
  }));

  // Stage 2: LLM refinement for ambiguous range
  const min = scoringCfg.llmAmbiguousMin ?? 30;
  const max = scoringCfg.llmAmbiguousMax ?? 70;
  const ambiguous = withScores.filter(i => i.score >= min && i.score <= max);

  if (scoringCfg.llmEnabled && ambiguous.length > 0) {
    console.log(`[scorer] Sending ${ambiguous.length} ambiguous items to LLM...`);
    const llmResults = await scoreBatchWithLLM(ambiguous, scoringCfg);
    for (const r of llmResults) {
      const item = withScores.find(i => i.url_hash === r.url_hash);
      if (item) {
        item.score = r.score;
        item.type = r.type;  // note: heuristics uses .type, llm returns .type
      }
    }
  }

  // Stage 3: Multi-source boost within this batch
  applyMultiSourceBoost(withScores);

  // Stage 4: Persist to DB
  const update = db.prepare(`
    UPDATE seen_urls
    SET score = ?, item_type = ?, threshold_pass = ?, scored_at = ?
    WHERE url_hash = ?
  `);

  const threshold = scoringCfg.threshold ?? 50;
  const now = Date.now();

  const persistBatch = db.transaction(items => {
    for (const item of items) {
      const pass = item.score >= threshold ? 1 : 0;
      update.run(item.score, item.type, pass, now, item.url_hash);
    }
  });

  persistBatch(withScores);

  const passed = withScores.filter(i => i.score >= threshold).length;
  console.log(
    `[scorer] Done — scored:${withScores.length} passed:${passed} threshold:${threshold}`
  );

  return { scored: withScores.length };
}
```

Note on field name: heuristics.js returns `{ score, type }` (not `item_type`). The `type` field is written to `item_type` column in the UPDATE. This is correct — `item_type` is the DB column name; `type` is the in-memory field name.
