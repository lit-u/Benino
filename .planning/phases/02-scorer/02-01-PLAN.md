```javascript
// Safe migration — SQLite does not support IF NOT EXISTS on ALTER TABLE
const addIfMissing = (col, type) => {

};
addIfMissing('score',          'INTEGER');
addIfMissing('item_type',      'TEXT');
```

```javascript
// scorer/test-scorer.js
// Smoke test for Phase 2 Scorer — run with: node scorer/test-scorer.js
// from: agent-network/server/services/news-collector/
import assert from 'assert';
import { createRequire } from 'module';

const require = createRequire(import.meta.url);
const config = require('./config.json');

let scoreByHeuristics, runScorer;

try {
  const hmod = await import('./scorer/heuristics.js');
  scoreByHeuristics = hmod.scoreByHeuristics;
} catch (e) {
  console.log('PENDING: scorer/heuristics.js not yet created');
}

try {
  const smod = await import('./scorer/index.js');
  runScorer = smod.runScorer;
} catch (e) {
  console.log('PENDING: scorer/index.js not yet created');
}

// SCOR-01 + SCOR-02: item gets score 0-100 and valid type
if (scoreByHeuristics) {
  const result = scoreByHeuristics({ title: 'Test item', url: 'https://example.com', source_id: 'hackernews' });
  assert(typeof result.score === 'number', 'SCOR-01: score must be a number');
  assert(result.score >= 0 && result.score <= 100, 'SCOR-01: score must be 0-100');
  assert(['breakthrough','release','update','research'].includes(result.type), 'SCOR-02: type must be valid');
  console.log('PASS SCOR-01+02: score=' + result.score + ' type=' + result.type);
}

// SCOR-03: GPT-5 launch scores >= 70
if (scoreByHeuristics) {
  const gpt5 = scoreByHeuristics({ title: 'OpenAI announces GPT-5 launch', url: 'https://openai.com/gpt-5', source_id: 'openai' });
  assert(gpt5.score >= 70, 'SCOR-03: GPT-5 launch must score >= 70, got ' + gpt5.score);
  console.log('PASS SCOR-03: GPT-5 score=' + gpt5.score);
}

// SCOR-04: multi-source boost
if (scoreByHeuristics) {
  const item1 = scoreByHeuristics({ title: 'Claude 4 released by Anthropic', url: 'https://anthropic.com/a', source_id: 'anthropic' });
  const item2 = scoreByHeuristics({ title: 'Claude 4 released by Anthropic', url: 'https://hackernews.com/b', source_id: 'hackernews' });
  const item3 = scoreByHeuristics({ title: 'Claude 4 released by Anthropic', url: 'https://github.com/c', source_id: 'github-trending' });
  // Multi-source boost is applied at runScorer level, not heuristics — note this
  console.log('INFO SCOR-04: will verify boost in runScorer integration test (Wave 3)');
}

// SCOR-05: threshold from config
const threshold = config.scoring?.threshold;
assert(typeof threshold === 'number', 'SCOR-05: config.scoring.threshold must exist');
console.log('PASS SCOR-05: threshold=' + threshold);

console.log('\nTest scaffold ready. Re-run after each wave to check PASS/PENDING counts.');
```
