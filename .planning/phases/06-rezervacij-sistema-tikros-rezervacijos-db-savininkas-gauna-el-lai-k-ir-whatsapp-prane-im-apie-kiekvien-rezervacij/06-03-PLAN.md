```javascript
function createReservationCard(res) {
  const card = document.createElement('div');
  card.className = `reservation-card status-${res.status}`;
  card.innerHTML = `
    <div class="res-header">
      <span class="res-room">${res.hotel_rooms?.name || 'Kambarys ' + (res.hotel_rooms?.number || '?')}</span>
      <span class="res-status-badge status-${res.status}">${STATUS_LABELS[res.status]}</span>
    </div>
    <div class="res-guest">${res.guest_name}</div>
    <div class="res-dates">${res.arrival_date} - ${res.departure_date} (${res.nights} nakt.)</div>
    ${res.total_price ? `<div class="res-price">${res.total_price} EUR</div>` : ''}
    <div>
      ${res.guest_email ? `<a href="mailto:${res.guest_email}" class="res-contact">${res.guest_email}</a>` : ''}
      ${res.guest_phone ? `<a href="tel:${res.guest_phone}" class="res-contact">${res.guest_phone}</a>` : ''}
    </div>
    ${res.message ? `<div class="res-message">${res.message}</div>` : ''}
    <div class="res-actions">
      ${res.status === 'pending' ? `<button class="btn-confirm" data-id="${res.id}">Patvirtinti</button>` : ''}
      ${res.status !== 'cancelled' ? `<button class="btn-cancel-res" data-id="${res.id}">Atsaukti</button>` : ''}
    </div>
  `;

  // Confirm handler
  const confirmBtn = card.querySelector('.btn-confirm');
  if (confirmBtn) {
    confirmBtn.addEventListener('click', () => {
      showConfirm('Patvirtinti sia rezervacija?', async () => {
        await updateReservationStatus(res.id, 'confirmed');
      });
    });
  }

  // Cancel handler
  const cancelBtn = card.querySelector('.btn-cancel-res');
  if (cancelBtn) {
    cancelBtn.addEventListener('click', () => {
      showConfirm('Atsaukti sia rezervacija?', async () => {
        await updateReservationStatus(res.id, 'cancelled');
      });
    });
  }

  return card;
}
```
