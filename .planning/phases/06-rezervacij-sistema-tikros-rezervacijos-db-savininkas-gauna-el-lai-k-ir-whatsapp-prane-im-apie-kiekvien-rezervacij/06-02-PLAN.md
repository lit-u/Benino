function validate(data) {
  if (!data.name || !data.arrival || !data.departure) return false;

**5. Add submit button click handler** (after the toggle button listener, replace the WhatsApp/email button handlers with):
```javascript
// Submit button — POST to API
const submitBtn = formEl.querySelector(`#btn-submit-${room.id}`);
if (submitBtn) {
  submitBtn.addEventListener('click', async () => {
    const data = getFormData();
    const errorEl = formEl.querySelector(`#form-error-${room.id}`);

    if (!validate(data)) {
      errorEl.textContent = !data.name || !data.arrival || !data.departure
        ? 'Uzpildykite visus privalomus laukus.'
        : 'Isvykimo data turi buti velesne nei atvykimo.';
      errorEl.classList.add('visible');
      return;
    }
    errorEl.classList.remove('visible');

    submitBtn.disabled = true;
    submitBtn.textContent = 'Siunčiama...';

    const slug = window.location.pathname.split('/hotel/')[1]?.split('/')[0]?.split('#')[0];
    const nights = Math.round((new Date(data.departure) - new Date(data.arrival)) / 86400000);
    const totalPrice = nights > 0 && room.price_per_night ? nights * room.price_per_night : null;

    try {
      const resp = await fetch(`/api/hotels/${encodeURIComponent(slug)}/reservations`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          room_id: room.id,
          guest_name: data.name,
          guest_email: data.email || null,
          guest_phone: data.phone || null,
          arrival_date: data.arrival,
          departure_date: data.departure,
          nights,
          total_price: totalPrice,
          message: data.message || null
        })
      });
      const json = await resp.json();
      if (json.success) {
        formEl.innerHTML = '<p class="reservation-success">Rezervacija priimta! Savininkas susisieks su jumis netrukus.</p>';
      } else {
        throw new Error(json.error || 'Server error');
      }
    } catch (err) {
      submitBtn.disabled = false;
      submitBtn.textContent = 'Siusti rezervacija';
      errorEl.textContent = 'Nepavyko issiusti. ' + (hotel.whatsapp ? 'Bandykite per WhatsApp.' : 'Bandykite veliau.');
      errorEl.classList.add('visible');
    }
  });
}
```
