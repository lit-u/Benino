    async function fetchSourceContent(url, fallbackDescription) {
      try {
        const res = await fetch(url, { signal: AbortSignal.timeout(8000), headers: { 'User-Agent': 'Mozilla/5.0' } });
        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const html = await res.text();
        const $ = cheerio.load(html);
        const text = $('article p, main p, .content p, p').map((_, el) => $(el).text().trim()).get().join(' ');
        if (text.length >= 200) return text;
      } catch (_) {}
      return fallbackDescription ?? '';
    }

    **Main export:**
    ```javascript
    export async function generateNewsPost(item, cfg) {
      const apiKey = process.env.OPENROUTER_API_KEY;
      if (!apiKey) throw new Error('OPENROUTER_API_KEY not set');

      const writerCfg = cfg.writer ?? {};
      const model = writerCfg.model ?? 'google/gemini-2.0-flash-001';
      const shortThreshold = writerCfg.shortPostWordThreshold ?? 200;
      const shortMaxTokens = writerCfg.shortPostMaxTokens ?? 700;
      const fullMaxTokens = writerCfg.fullPostMaxTokens ?? 1200;

      const client = new OpenAI({ baseURL: 'https://openrouter.ai/api/v1', apiKey });

      const content = await fetchSourceContent(item.url, item.description);
      const wordCount = content.split(/\s+/).filter(Boolean).length;
      const maxTokens = wordCount < shortThreshold ? shortMaxTokens : fullMaxTokens;

      const [moksliusBody, oldboyBody, tags] = await Promise.all([
        generateMokslius(item.title, content, client, model, maxTokens),
        generateOldBoy(item.title, content, client, model, maxTokens),
        generateTags(item.title, content, client, model),
      ]);

      const category = detectCategory(item.source_id, item.item_type);

      return {
        title: item.title,
        moksliusBody,
        oldboyBody,
        tags: Array.isArray(tags) ? tags.slice(0, 5) : [],
        category,
      };
    }
    ```

    Note: three LLM calls run in parallel via Promise.all to minimize latency.
