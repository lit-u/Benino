    ```javascript
    /**
     * News moderation routes.
     * POST /api/news/accept/:hash — Writer + Publisher pipeline (Phase 4)
     */
    import express from 'express';
    import { createClient } from '@supabase/supabase-js';
    import { createRequire } from 'module';
    import { fileURLToPath } from 'url';
    import path from 'path';
    import { initDb, setWriterStatus } from '../services/news-collector/storage/db.js';
    import { generateNewsPost } from '../services/news-collector/writer/news-writer.js';
    import { publishNewsPost } from '../services/news-collector/writer/publisher.js';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    const require = createRequire(import.meta.url);
    const cfg = require('../services/news-collector/config.json');

    const router = express.Router();

    // Supabase client for blog_posts insert
    const supabase = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_KEY ||
      process.env.SUPABASE_SERVICE_ROLE_KEY ||
      process.env.SUPABASE_KEY
    );

    /**
     * POST /api/news/accept/:hash
     * Triggered by Telegram bot handleAccept() when admin presses ✅.
     * Pipeline: load item → set writer_status='writing' → generateNewsPost → publishNewsPost → set 'published'
     * Returns: { success: true, url: 'http://localhost:3000/blog/slug' }
     */
    router.post('/accept/:hash', async (req, res) => {
      const { hash } = req.params;
      const db = initDb();

      try {
        // 1. Load item from DB
        const item = db.prepare('SELECT * FROM seen_urls WHERE url_hash = ?').get(hash);
        if (!item) {
          return res.status(404).json({ error: 'Item not found', hash });
        }

        // 2. Idempotency guard — prevent double-publish
        if (item.writer_status === 'writing' || item.writer_status === 'published') {
          return res.status(409).json({
            error: 'Conflict',
            message: `Item already in writer_status: ${item.writer_status}`,
            hash,
          });
        }

        // 3. Mark as in-progress
        setWriterStatus(db, hash, 'writing');

        // 4. Generate post via LLM
        console.log(`[news-route] Generating post for: ${item.title}`);
        const post = await generateNewsPost(item, cfg);

        // 5. Publish to blog_posts
        const { slug, url } = await publishNewsPost(post, supabase);

        // 6. Mark as published
        setWriterStatus(db, hash, 'published');

        console.log(`[news-route] Published: ${url}`);
        return res.status(200).json({ success: true, url, slug, title: post.title });

      } catch (err) {
        console.error(`[news-route] accept failed for ${hash}:`, err.message);
        try { setWriterStatus(db, hash, 'failed'); } catch (_) {}
        return res.status(500).json({ error: 'Writer pipeline failed', message: err.message, hash });
      } finally {
        db.close();
      }
    });

    export default router;
    ```

    The `db.close()` in finally matches the established Phase 01-02 pattern of closing DB after each route operation. initDb() opens a fresh connection — safe because news.js is called infrequently (once per admin accept action).
