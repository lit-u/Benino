import { Telegraf } from 'telegraf';

if (!process.env.TELEGRAM_BOT_TOKEN) {
  console.warn('[telegram] TELEGRAM_BOT_TOKEN not set — bot will not start');
}

const bot = process.env.TELEGRAM_BOT_TOKEN
  ? new Telegraf(process.env.TELEGRAM_BOT_TOKEN)
  : null;

export { bot };

export function startBotPolling() {
  if (!bot) {
    console.warn('[telegram] Skipping bot.launch() — no token configured');
    return;
  }
  bot.launch();
  process.once('SIGINT',  () => bot.stop('SIGINT'));
  process.once('SIGTERM', () => bot.stop('SIGTERM'));
  console.log('[telegram] Bot polling started');
}

import { Markup } from 'telegraf';

const TYPE_EMOJI = {
  breakthrough: '🚀',
  release: '📦',
  update: '🔧',
  research: '📄',
};

/**
 * Escape all MarkdownV2 reserved characters.
 * Required chars per Telegram Bot API docs:
 * _ * [ ] ( ) ~ ` > # + - = | { } . ! \
 */
export function escapeMarkdownV2(text) {
  return String(text).replace(/[_*[\]()~`>#+\-=|{}.!\\]/g, '\\$&');
}

/**
 * Build a Telegram card for a news item.
 * @param {object} item - Row from seen_urls: { url_hash, title, source_id, score, item_type, url }
 * @returns {{ text: string, keyboard: object }}
 */
export function buildCard(item) {
  const emoji  = TYPE_EMOJI[item.item_type] ?? '📰';
  const score  = item.score ?? '?';
  const title  = escapeMarkdownV2(item.title ?? 'Untitled');
  const source = escapeMarkdownV2(item.source_id ?? 'unknown');
  const type   = escapeMarkdownV2(item.item_type ?? 'update');
  const url    = item.url ?? '';

  const text =
    `*${title}*\n` +
    `Score: ${score} \\| ${emoji} ${type}\n` +
    `📰 ${source}\n` +
    `[Skaityti daugiau](${url})`;

  const keyboard = Markup.inlineKeyboard([
    Markup.button.callback('✅ Priimti',  `accept:${item.url_hash}`),
    Markup.button.callback('❌ Atmesti', `reject:${item.url_hash}`),
  ]);

  return { text, keyboard };
}

/**
 * Register bot.action() handlers for Accept and Reject inline buttons.
 *
 * @param {import('telegraf').Telegraf} bot
 * @param {import('better-sqlite3').Database} db  - persistent bot-lifetime connection
 */
export function registerHandlers(bot, db) {
  bot.action(/^reject:(.+)$/, async (ctx) => {
    const urlHash = ctx.match[1];
    try {
      db.prepare("UPDATE seen_urls SET tg_status='rejected' WHERE url_hash=?").run(urlHash);
      await ctx.editMessageText('❌ Atmesta', { reply_markup: { inline_keyboard: [] } });
      await ctx.answerCbQuery('Atmesta');
    } catch (err) {
      console.error('[telegram] reject handler error:', err.message);
      try { await ctx.answerCbQuery('Klaida'); } catch (_) {}
    }
  });

  bot.action(/^accept:(.+)$/, async (ctx) => {
    const urlHash = ctx.match[1];
    try {
      db.prepare("UPDATE seen_urls SET tg_status='accepted' WHERE url_hash=?").run(urlHash);
      await ctx.editMessageText('⏳ Generuojama\\.\\.\\.', {
        parse_mode: 'MarkdownV2',
        reply_markup: { inline_keyboard: [] },
      });
      await ctx.answerCbQuery('Priimta');
    } catch (err) {
      console.error('[telegram] accept handler error:', err.message);
      try { await ctx.answerCbQuery('Klaida'); } catch (_) {}
    }

    // Phase 4 stub — trigger writer pipeline
    try {
      const port = process.env.PORT || 3000;
      const res = await fetch(`http://localhost:${port}/api/news/accept/${urlHash}`, { method: 'POST' });
      if (!res.ok) {
        console.warn(`[telegram] accept stub returned ${res.status} for ${urlHash}`);
      }
    } catch (err) {
      // Phase 4 not yet implemented — do not crash
      console.warn('[telegram] accept stub call failed (Phase 4 pending):', err.message);
    }
  });
}

```javascript
import { createRequire } from 'module';
import { buildCard } from './card.js';
import { bot } from './bot.js';

const require = createRequire(import.meta.url);
const config  = require('../config.json');

const adminChatId     = process.env.TELEGRAM_ADMIN_CHAT_ID;
const batchSize       = config.telegram?.batchSize       ?? 10;
const batchDelayMs    = config.telegram?.batchDelayMs    ?? 1000;
const cutoffMs        = config.telegram?.dispatchCutoffMs ?? 86400000;  // 24h

/**
 * Query all threshold-passing items not yet dispatched and send each as a Telegram card.
 * Only items scored within the last cutoffMs are sent (avoids flooding admin with stale backlog).
 *
 * @param {import('better-sqlite3').Database} db
 * @returns {Promise<{sent: number, skipped: number, errors: number}>}
 */
export async function runTelegramDispatch(db) {
  if (!bot) {
    console.warn('[telegram] runTelegramDispatch: no bot configured, skipping');
    return { sent: 0, skipped: 0, errors: 0 };
  }
  if (!adminChatId) {
    console.warn('[telegram] runTelegramDispatch: TELEGRAM_ADMIN_CHAT_ID not set, skipping');
    return { sent: 0, skipped: 0, errors: 0 };
  }

  const cutoffTs = Date.now() - cutoffMs;

  const pending = db.prepare(`
    SELECT url_hash, url, title, source_id, score, item_type
    FROM seen_urls
    WHERE threshold_pass = 1
      AND (tg_status IS NULL)
      AND scored_at > ?
    ORDER BY score DESC
    LIMIT ?
  `).all(cutoffTs, batchSize);

  console.log(`[telegram] dispatch: ${pending.length} items to send`);

  const results = { sent: 0, skipped: 0, errors: 0 };

  for (const item of pending) {
    try {
      const { text, keyboard } = buildCard(item);
      const msg = await bot.telegram.sendMessage(adminChatId, text, {
        parse_mode: 'MarkdownV2',
        ...keyboard,
        disable_web_page_preview: true,
      });
      db.prepare(
        "UPDATE seen_urls SET tg_status='pending', tg_message_id=? WHERE url_hash=?"
      ).run(msg.message_id, item.url_hash);
      results.sent++;

      // Rate limiting: 1 message per second
      if (results.sent < pending.length) {
        await new Promise(resolve => setTimeout(resolve, batchDelayMs));
      }
    } catch (err) {
      console.error(`[telegram] failed to send card for ${item.url_hash}:`, err.message);
      results.errors++;
    }
  }

  console.log(`[telegram] dispatch done — sent:${results.sent} errors:${results.errors}`);
  return results;
}
```

Key decisions in this implementation:
- `AND (tg_status IS NULL)` — only unsent items (not 'pending', 'accepted', 'rejected')
- `AND scored_at > cutoffTs` — avoids flooding with 942-item backlog from Phases 1+2
- After send: sets tg_status='pending' (item is in Telegram, awaiting admin decision)
- `disable_web_page_preview: true` — keeps card clean without URL preview
- setTimeout between sends respects Telegram's rate limits
